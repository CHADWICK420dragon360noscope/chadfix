# Viking Raider
unit_event = {
	id = 700
	hide_window = yes
	
	ai = no
	
	is_triggered_only = yes
	
	trigger = {
		has_dlc = "Charlemagne"
		FROM = {
			culture_group = north_germanic
			OR = {
				religion = norse_pagan
				religion = norse_pagan_reformed
			}
		}
		loot = 1000
	}
	
	immediate = {
		FROM = {
			if = {
				limit = {
					NOT = { has_global_flag = achievement_viking_raider }
				}
				set_global_flag = achievement_viking_raider
				chronicle = {
					entry = CHRONICLE_ACHIEVEMENT_VIKING_RAIDER

				narrative_event = {
					id = chronicled_achievements.1

	}
}

# Full House
character_event = {
	id = 706
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		has_dlc = "Charlemagne"
		OR = {
			AND = {
				NOT = {
					has_global_flag = achievement_full_house
				}
				OR = {
					father = {
						ai = no
						NOT = { has_character_flag = achievement_full_house }
					}
					mother = { 
						ai = no 
						NOT = { has_character_flag = achievement_full_house }
					}
				}
			}
			AND = {
				NOT = {
					has_global_flag = achievement_cheaper_dozen
				}
				OR = {
					father = {
						ai = no
						NOT = { has_character_flag = achievement_cheaper_dozen }
					}
					mother = { 
						ai = no 
						NOT = { has_character_flag = achievement_cheaper_dozen }
					}
				}
			}
		}
	}
	
	immediate = {
		
		if = {
			limit = {
				father = { 
					ai = no
					NOR = {
						has_character_flag = achievement_cheaper_dozen
						check_variable = { which = "bob_saget" value = 11.5 }
					}
				}
			}
			father = {
				change_variable = { which = "bob_saget" value = 1 }
			}
		}
		if = {
			limit = {
				mother = { 
					ai = no
					NOR = {
						has_character_flag = achievement_cheaper_dozen
						check_variable = { which = "bob_saget" value = 4.5 }
					}
				}
			}
			mother = {
				change_variable = { which = "bob_saget" value = 1 }
			}
		}
		
		if = {

					has_global_flag = achievement_cheaper_dozen

					ai = no
					NOT = { has_character_flag = achievement_cheaper_dozen }
					check_variable = { which = "bob_saget" value = 11.5 }
				}
			}
			father = {
				set_character_flag = achievement_cheaper_dozen
				set_global_flag = achievement_cheaper_dozen
				chronicle = {
					entry = CHRONICLE_ACHIEVEMENT_CHEAPER_DOZEN
					portrait = [This.GetID]
					picture = GFX_evt_birth
				}
				narrative_event = {
					id = chronicled_achievements.160
				}
			}
			break = yes
		}
		if = {
			limit = {
				has_global_flag = achievement_full_house
				mother = { 
					ai = no
					NOT = { has_character_flag = achievement_full_house }
					check_variable = { which = "bob_saget" value = 4.5 }
				}
			}
			mother = {
				set_character_flag = achievement_full_house
